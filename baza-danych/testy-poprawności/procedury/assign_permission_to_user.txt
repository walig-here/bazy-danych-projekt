SCENARIUSZ 1 -	poprawna przypisanie użytkownikowi uprawnienia.

----------------------------------------------------------------------------------
CALL assign_permission_to_user(
	(SELECT au.administrative_unit_id FROM administrative_units AS au WHERE `name` = 'Dolnośląskie' LIMIT 1),
	(SELECT u.login FROM registered_users AS u WHERE u.`role` = 'meritorical_administrator' LIMIT 1),
	1
);
SELECT * FROM granted_permissions;
----------------------------------------------------------------------------------

Poprawie nadano administratorowi merytorycznemu uprawnienie do edytowania 
mejscowości w województwie Dolnośląskim.
